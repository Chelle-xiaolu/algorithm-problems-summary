// 二分查找实现 当mid*mid<=x时就是要找的答案
// 所有的元素都是整数运算，不会存在误差
var mySqrt = function(x) {
  let left = 0, right = x, ans = -1
  while(left <= right) {
    const mid = left + Math.floor((right - left)/2)
    if(mid*mid > x) {
      right = mid - 1
    } else {
      ans = mid
      left = mid+1
    }
  }
  return ans
};

```
/**
 * 牛顿迭代法 算法原理：用(x,f(x))的切线来不断逼近x*2-a=0的根
 * 函数上任意一点的切线斜率都是2x
 * x-f(x)/(2x)就是一个比x更接近的近似值 代入f(x)=x*2-a
 * x - (x*x-a)/(2x) = x-x/2+a/2x = (x+a/x)/2
 */
const mySqrt = (x) => {
  let r = x
  while(r*r > x) {
    r = Math.floor(0.5*(r+x/r))
  }
  return r
}

3. leetcode【33】搜索旋转排序数组
```
/**
 *怎么寻找这种数组中的有序数组？
 *思路一：暴力还原（O(logn)）+二分
 *思路二： 直接二分 
 *nums[mid] > nums[0] 左边有序 判断target是否在左区间内，在的话right=mid-1 否则left=mid+1
 *nums[mid] < nums[0] 右边有序 判断target是否在区间内，在的话left=mid+1 否则right=mid-1
 */
var search = function(nums, target) {
  const len = nums.length
  if(!len) {
    return -1
  }
  if(len === 1) {
    return nums.indexOf(target)
  }
  let left = 0, right = len-1
  while(left <= right) {
    let mid = Math.floor(left+(right-left)/2)
    if(nums[mid] === target) {
      return mid
    }
    if(nums[mid] >= nums[left]) {
      if(target >= nums[left] && target < nums[mid]) {
        right = mid-1
      } else {
        left = mid+1
      }
    } else {
      if(target > nums[mid] && target <= nums[right]){
        left = mid+1
      } else {
        right = mid-1
      }
    }
  }
  return -1
};

4. leetcode【153】寻找旋转排序数组中的最小值--数组无重复元素
```
// 典型二分，注意临界值
var findMin = function(nums) {
  let left = 0, right = nums.length-1
  while(left <= right) {
    if(nums[left]<=nums[right]){
      return nums[left]
    }
    let mid = Math.floor(left+(right-left)/2)
    if(nums[mid] >= nums[left]) {
      // 左边有序单调的 变化点就在右边
      left = mid+1
    } else {
      // 这里right=mid mid可能刚刚好就是最小值，不能划分在区间外
      right = mid
    }
  }
  return nums[left]
};

5. leetcode【154】寻找旋转排序数组中的最小值II-有重复元素
```
// 二分查找
var findMin = function(nums) {
  let left = 0, right = nums.length-1
  let mid = 0
  while(left <= right) {
    // 二分后的数组是有序数组，直接返回nums[left]
    if(nums[left] < nums[right]) {
      return nums[left]
    }
    mid = Math.floor(left+(right-left)/2)
    if(nums[mid] > nums[left]) {
      // 中间元素大于left 那么最小值一定在右边界
      left = mid+1
    } else if(nums[mid] === nums[left]) {
      // 有重复元素时left++
      left++
    } else {
      // 中间元素小于左边界，最小元素一定在最左和mid之间，并且不是左边界
      right = mid
      left++
    }
  }
  return nums[mid]
};

6. leetcode【74】搜索二维矩阵
```
// 思路一：转化为一维数组 二分查找 时间复杂度O(mn)
// cx = mid / n cy = mid % n
var searchMatrix = function(matrix, target) {
  if(!matrix.length) {
    return false
  }
  const m = matrix.length, n = matrix[0].length
  let left = 0, right = m*n-1
  while(left <= right) {
    let mid = Math.floor(left+(right-left)/2)
    let cx = Math.floor(mid/n), cy = mid%n
    if(matrix[cx][cy] === target) {
      return true
    }
    if(matrix[cx][cy] > target) {
      right = mid - 1
    } else {
      left = mid + 1
    }
  }
  return false
};
```
